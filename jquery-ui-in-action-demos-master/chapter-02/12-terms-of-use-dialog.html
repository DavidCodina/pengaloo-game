Terms of Use Dialog:


	$("#button1")
		.wrap("<div class='text-center my-4 p-4'></div>")
		.on('click', function() {
			$( "#dialog" ).dialog({
				buttons: {
					OK: function() {
						$( this ).dialog( "close" );
					}
				},

				beforeClose: function( event, ui ) {
					if ( !$( "#terms" ).prop( "checked" ) ) {
						event.preventDefault();
						$( "[for=terms]" ).addClass( "ui-state-error-text" );
					}
				},

				width: 600
			});
	});
