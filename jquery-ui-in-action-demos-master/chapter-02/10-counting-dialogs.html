Counting Dialogs:

In this example, I also added a form to make custom messages.


	//Make a dialog with a custom title and content.
	$("#form1").on("submit", function(event){
		event.preventDefault();

		var title      = $("input[name='title']").val();
		var message    = $("input[name='message']").val();
		var new_dialog = $("<div>" + message + "</div>").dialog({ title: title });
	});


	//Or just make a blank dialog.
	$("#button1")
		.wrap("<div class='text-center my-4 p-4'></div>")
		.on('click', function(){ $("<div>").dialog(); });


	//Count the number of dialogs made.
	var dialogs = 0;

	$( document ).on( "dialogcreate", function() {
		dialogs++;
		$( "#count" ).text( dialogs );
	});
